                                      totalQty:=ILE.Quantity;
                                      //Total Value

                                      ValueEntry.SETRANGE("Item No.","Item Ledger Entry"."Item No.");
                                      ValueEntry.SETFILTER("Posting Date",'..%1',myDate);
                                      ValueEntry.CALCSUMS("Cost Amount (Actual)");
                                      totalValue:=ValueEntry."Cost Amount (Actual)";

                                      //<30Days
                                      ILE.SETFILTER("Posting Date",'%1..%2',CALCDATE('-30d',myDate),CALCDATE('0d',myDate));
                                      ILE.CALCSUMS(Quantity);
                                      qty_30:=ILE.Quantity;

                                      ValueEntry.SETFILTER("Posting Date",'%1..%2',CALCDATE('-30d',myDate),CALCDATE('0d',myDate));
                                      ValueEntry.CALCSUMS("Cost Amount (Actual)");
                                      value_30:=ValueEntry."Cost Amount (Actual)";

                                      //30-60Days
                                      ILE.SETFILTER("Posting Date",'%1..%2',CALCDATE('-60d',myDate),CALCDATE('-30d',myDate));
                                      ILE.CALCSUMS(Quantity);

                                      ValueEntry.SETFILTER("Posting Date",'%1..%2',CALCDATE('-60d',myDate),CALCDATE('-30d',myDate));

                                      ValueEntry.SETFILTER("Posting Date",'%1..%2',CALCDATE('-90d',myDate),CALCDATE('-60d',myDate));
                                      ValueEntry.CALCSUMS("Cost Amount (Actual)");
                                      value_90:=ValueEntry."Cost Amount (Actual)";


                                      //90-180Days
                                      ILE.SETFILTER("Posting Date",'%1..%2',CALCDATE('-180d',myDate),CALCDATE('-90d',myDate));
                                      ILE.CALCSUMS(Quantity);
                                      qty_180:=ILE.Quantity;

                                      ValueEntry.SETFILTER("Posting Date",'%1..%2',CALCDATE('-180d',myDate),CALCDATE('-90d',myDate));
                                      ValueEntry.CALCSUMS("Cost Amount (Actual)");
                                      value_180:=ValueEntry."Cost Amount (Actual)";


                                      //180-365Days
                                      ILE.SETFILTER("Posting Date",'%1..%2',CALCDATE('-365d',myDate),CALCDATE('-180d',myDate));
                                      ILE.CALCSUMS(Quantity);
                                      qty_365:=ILE.Quantity;

                                      ValueEntry.SETFILTER("Posting Date",'%1..%2',CALCDATE('-365d',myDate),CALCDATE('-180d',myDate));
                                      ValueEntry.CALCSUMS("Cost Amount (Actual)");
                                      value_365:=ValueEntry."Cost Amount (Actual)";


                                      //2Y
                                      ILE.SETFILTER("Posting Date",'%1..%2',CALCDATE('-730d',myDate),CALCDATE('-365d',myDate));
                                      ILE.CALCSUMS(Quantity);
                                      qty_2Y:=ILE.Quantity;

                                      ValueEntry.SETFILTER("Posting Date",'%1..%2',CALCDATE('-730d',myDate),CALCDATE('-365d',myDate));
                                      ValueEntry.CALCSUMS("Cost Amount (Actual)");
                                      value_2Y:=ValueEntry."Cost Amount (Actual)";


                                      //3Y
                                      ILE.SETFILTER("Posting Date",'%1..%2',CALCDATE('-1095d',myDate),CALCDATE('-730d',myDate));
                                      ILE.CALCSUMS(Quantity);
                                      qty_3Y:=ILE.Quantity;

                                      ValueEntry.SETFILTER("Posting Date",'%1..%2',CALCDATE('-1095d',myDate),CALCDATE('-730d',myDate));

                                      //>3Y
                                      ILE.SETFILTER("Posting Date",'<%1',CALCDATE('-1095d',myDate));
                                      ILE.CALCSUMS(Quantity);
                                      qty_3YM:=ILE.Quantity;

                                      ValueEntry.SETFILTER("Posting Date",'<%1',CALCDATE('-1095d',myDate));
                                      ValueEntry.CALCSUMS("Cost Amount (Actual)");
                                      value_3YM:=ValueEntry."Cost Amount (Actual)";





                                      Srno+=1;
                                    END
                                  ELSE
                                    BEGIN
                                      itemNo:="Item Ledger Entry"."Item No.";
                                      CurrReport.SKIP;
                                    END;

                                END;
